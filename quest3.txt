#include <stdio.h>
#include <math.h>

int main() {
  double x, y;
  
  if (scanf("%lf", &x) != 1) {
    printf("n/a");
    return 0;
  }
  
  if (x == 0 || x * x / 2 == 0) {
    printf("n/a");
    return 0;
  }
  
  y = 0.007 * pow(x, 4) 
      + ((22.8 * cbrt(x) - 1000) * x + 3) / (x * x / 2) 
      - x * pow(10 + x, 2.0 / x) 
      - 1.01;
  
  printf("%.1f", y);
  
  return 0;
}
